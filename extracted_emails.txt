import re

with open('extracted_emails.txt', 'r') as f:
    lines = f.readlines()

emails = []
for line in lines:
    # Split by colon to separate URL from emails, take the second part
    if ':' in line:
        email_part = line.split(':', 1)[1].strip()
        # Some lines might have multiple emails separated by commas
        parts = [e.strip() for e in email_part.split(',')]
        emails.extend(parts)

# Remove duplicates while preserving order (optional, but good practice, though user might just want the raw list)
# Let's just output the unique emails
unique_emails = list(dict.fromkeys(emails))

print('\n'.join(unique_emails))
